#Clean R session before running next script
clear_R <- function (){

	rm(list = x[!(x%in% GLOBAL)],pos=1) # Clear variables not in GLOBAL
	sapply(file.path(tempdir(), list.files(tempdir())), unlink)
	gc()
}
